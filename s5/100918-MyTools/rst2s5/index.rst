.. container:: handout

    云端序列图谱生成服务... |zqeye|


.. code-block:: python

    # wsd_flow.txt
        participant "思想" as mind
        participant "脚本" as txt
        participant "WebSequenceDiagrams.com" as WSD

        mind->txt: 记述思想
        activate txt
        txt->WSD: 直接提交
        activate WSD
        WSD->mind: 返回图片
        deactivate WSD
        mind->mind: 对话/重构
        deactivate mind
        mind->txt: 修订记述
