        const OptionsPage = options.OptionsPage;
        function MouseGestureOptions() {	
            OptionsPage.call(this, 'mousegesture',
                '选项 - 鼠标手势','mousegesturePage');
        };
        MouseGestureOptions.prototype = {
             __proto__: options.OptionsPage.prototype,
